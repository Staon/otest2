46 void AssertContext::enterAssertion(
47  bool condition_,
48  const std::string& message_,
49  bool use_expression_) {
50  /* -- change the result of current object */
51  if(!condition_)
52  context->semantic_stack->setTop(false);
53 
54  /* -- report the assertion */
55  std::ostringstream sos_;
56  if(use_expression_ && !expression.empty())
57  sos_ << '\'' << expression << '\'';
58  if(!message_.empty()) {
59  if(use_expression_ && !expression.empty())
60  sos_ << " ";
61  sos_ << message_;
62  }
63  context->reporter->enterAssert(
64  *context, condition_, sos_.str(), file, lineno);
65 }
