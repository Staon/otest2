40 template<template<typename, typename> class Compare_, typename IterA_, typename IterB_>
41 bool compareListsLexicographically(
42  std::vector<std::string>& messages_,
43  IterA_ begin_a_,
44  IterA_ end_a_,
45  IterB_ begin_b_,
46  IterB_ end_b_) {
47  typedef typename AssertionParameter<decltype(*begin_a_)>::Type AType_;
48  typedef typename AssertionParameter<decltype(*begin_b_)>::Type BType_;
49  typedef Compare_<AType_, BType_> Comparator_;
50 
51  int index_(0);
52  Comparator_ cmp_;
53  std::ostringstream sos_;
54  while(begin_a_ != end_a_ && begin_b_ != end_b_) {
55  int cmp_result_(cmp_.checkItems(*begin_a_, *begin_b_));
56  if(cmp_result_ < 0) {
57  /* -- the a_ list is lesser */
58  sos_ << "the assertion 'a ";
59  PrintTrait<Comparator_>::print(sos_, cmp_);
60  sos_ << " b' has passed";
61  messages_.push_back(sos_.str());
62  return true;
63  }
64  if(cmp_result_ > 0) {
65  /* -- the a_ list is greater */
66  sos_ << "the ";
67  printOrdinalNumber(sos_, index_);
68  sos_ << " items have failed: a[" << index_ << "] ";
69  PrintTrait<Comparator_>::print(sos_, cmp_);
70  sos_ << " b[" << index_ << "]";
71  messages_.push_back(sos_.str());
72 
73  sos_.str("");
74  sos_ << "a[" << index_ << "] = ";
75  PrintTrait<AType_>::print(sos_, *begin_a_);
76  messages_.push_back(sos_.str());
77 
78  sos_.str("");
79  sos_ << "b[" << index_ << "] = ";
80  PrintTrait<AType_>::print(sos_, *begin_b_);
81  messages_.push_back(sos_.str());
82 
83  return false;
84  }
85 
86  /* -- move to next item */
87  ++begin_a_;
88  ++begin_b_;
89  ++index_;
90  }
91 
92  /* -- All items equal each other in the pair. The result is defined by
93  * length of resting lists - whether a list is longer then the other
94  * or shorter or the same length. */
95  const bool a_ends_(begin_a_ == end_a_);
96  const bool b_ends_(begin_b_ == end_b_);
97  if(cmp_.checkRests(a_ends_, b_ends_)) {
98  sos_ << "the assertion 'a ";
99  PrintTrait<Comparator_>::print(sos_, cmp_);
100  sos_ << " b' has passed: one list is the prefix of the other one";
101  messages_.push_back(sos_.str());
102  return true;
103  }
104 
105  /* -- The lists doesn't match the operator */
106  sos_ << "the assertion 'a ";
107  PrintTrait<Comparator_>::print(sos_, cmp_);
108  sos_ << " b' has failed.";
109  messages_.push_back(sos_.str());
110 
111  if(a_ends_ && b_ends_) {
112  messages_.push_back("the lists are the same length");
113  }
114  else if(a_ends_) {
115  messages_.push_back("the list a is shorter");
116  }
117  else {
118  messages_.push_back("the list a is longer");
119  }
120  return false;
121 }

125 template<template<typename, typename> class Compare_, typename IterA_, typename IterB_>
126 bool LexicographicalAssertion::testAssertLexiList(
127  IterA_ begin_a_,
128  IterA_ end_a_,
129  IterB_ begin_b_,
130  IterB_ end_b_) {
131  /* -- check the container */
132  std::vector<std::string> messages_;
133  bool result_(Private::compareListsLexicographically<Compare_>(
134  messages_, begin_a_, end_a_, begin_b_, end_b_));
135  assert(messages_.size() > 0);
136 
137  /* -- report the result */
138  enterAssertion(result_, messages_[0], false);
139  for(int i_(1); i_ < messages_.size(); ++i_)
140  assertionMessage(result_, messages_[i_]);
141  return leaveAssertion(result_);
142 }
