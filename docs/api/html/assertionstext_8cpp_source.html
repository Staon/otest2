51  oss_ << std::setfill('0');

74 bool LongTextAssertion::testAssertImpl(
75  std::istream& a_,
76  std::istream& b_) {
77  /* -- read contents of the files */
78  std::vector<std::string> a_data_;
79  slurpFile(a_data_, a_);
80  std::vector<std::string> b_data_;
81  slurpFile(b_data_, b_);
82 
83  /* -- compute the difference */
84  DiffLogBlocks diff_log_;
85  DiffLogBuilderBlock log_builder_(&diff_log_);
86  hirschbergDiff(a_data_, b_data_, log_builder_);
87 
88  bool result_(diff_log_.empty());
89  if(result_) {
90  /* -- There is no difference, the files match */
91  enterAssertion(result_, "the texts are equal", false);
92  }
93  else {
94  /* -- the files are different */
95  enterAssertion(result_, "the texts are different", false);
96 
97  /* -- print the difference */
98  constexpr int CONTEXT(3);
99  bool trailing_context_(false);
100  int left_line_(0);
101  int right_line_(0);
102  std::ostringstream oss_;
103  for(const auto& difference_ : diff_log_) {
104  /* -- print trailing context of previous change */
105  if(trailing_context_) {
106  const int context_end_(left_line_ + CONTEXT);
107  for(;
108  left_line_ < difference_.left_begin && left_line_ < context_end_;
109  ++left_line_, ++right_line_) {
110  assert(right_line_ < difference_.right_begin);
111 
112  assertionMessage(
113  result_,
114  formatLine(left_line_, right_line_, ' ', a_data_[left_line_]));
115  }
116  }
117  else
118  trailing_context_ = true;
119 
120  /* -- skip unchanged lines */
121  if(left_line_ < difference_.left_begin - CONTEXT) {
122  assert(right_line_ < difference_.right_begin);
123 
124  assertionMessage(result_, "........................................");
125  left_line_ = difference_.left_begin - CONTEXT;
126  right_line_ = difference_.right_begin - CONTEXT;
127  }
128 
129  /* -- print entering context of the change */
130  for(;
131  left_line_ < difference_.left_begin;
132  ++left_line_, ++right_line_) {
133  assert(right_line_ < difference_.right_begin);
134 
135  assertionMessage(
136  result_,
137  formatLine(left_line_, right_line_, ' ', a_data_[left_line_]));
138  }
139 
140  /* -- print the difference */
141  assert(left_line_ == difference_.left_begin);
142  assert(right_line_ == difference_.right_begin);
143  for(; left_line_ < difference_.left_end; ++left_line_) {
144  assertionMessage(
145  result_,
146  formatLine(left_line_, -1, '<', a_data_[left_line_]));
147  }
148  for(; right_line_ < difference_.right_end; ++right_line_) {
149  assertionMessage(
150  result_,
151  formatLine(-1, right_line_, '>', b_data_[right_line_]));
152  }
153  }
154 
155  /* -- print ending context */
156  const int context_end_(left_line_ + CONTEXT);
157  for(;
158  left_line_ < a_data_.size() && left_line_ < context_end_;
159  ++left_line_, ++right_line_) {
160  assert(right_line_ < b_data_.size());
161 
162  assertionMessage(
163  result_,
164  formatLine(left_line_, right_line_, ' ', a_data_[left_line_]));
165  }
166 
167  /* -- skipt the end of the file */
168  if(left_line_ < a_data_.size()) {
169  assert(right_line_ < b_data_.size());
170 
171  assertionMessage(result_, "........................................");
172  }
173  }
174  return leaveAssertion(result_);
175 }
