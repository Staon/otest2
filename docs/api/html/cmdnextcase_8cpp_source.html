57 void CmdNextCase::run(
58  const Context& context_) {
59  /* -- get the case factory */
60  std::string case_name_;
61  CaseFactoryPtr factory_(suite->getCase(context_, current, &case_name_));
62  if(factory_ != nullptr) {
63  /* -- There is a case at the index. Schedule run of next one. */
64  context_.command_stack->pushCommand(
65  std::make_shared<CmdNextCase>(suite, current + 1));
66 
67  /* -- create and schedule the test-case if it's not filtered */
68  if(!context_.runner_filter->filterCase(suite->getName(), case_name_)) {
69  /* -- prepare the stack frame */
70  context_.semantic_stack->push(true);
71  context_.object_path->pushName(case_name_);
72 
73  /* -- report entering of the test case */
74  context_.reporter->enterCase(context_, case_name_);
75 
76  /* -- schedule finishing of the test case */
77  context_.command_stack->pushCommand(std::make_shared<CmdLeaveCase>());
78 
79  /* -- Create and schedule the test case. As the constructor method
80  * can throw an exception the code is run in the protected
81  * environment. */
82  runUserCode(context_, [&](const Context& context_){
83  CasePtr testcase_(factory_->createCase(context_));
84  testcase_->scheduleRun(context_, testcase_);
85  });
86  }
87  }
88 }
