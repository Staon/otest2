52 void CmdNextCase::run(
53  const Context& context_) {
54  /* -- get the case factory */
55  std::string case_name_;
56  CaseFactoryPtr factory_(suite->getCase(context_, current, &case_name_));
57  if(factory_ != nullptr) {
58  /* -- There is a case at the index. Schedule run of next one. */
59  context_.command_stack->pushCommand(
60  std::make_shared<CmdNextCase>(suite, current + 1));
61 
62  /* -- Schedule run of the test case */
63  if(!context_.runner_filter->filterCase(suite->getName(), case_name_)) {
64  CasePtr testcase_(factory_->createCase(context_));
65  testcase_->scheduleRun(context_, testcase_);
66  }
67  }
