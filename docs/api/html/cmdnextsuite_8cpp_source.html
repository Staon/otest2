56 void CmdNextSuite::run(
57  const Context& context_) {
58  /* -- create the suite */
59  std::string suite_name_;
60  SuiteFactoryPtr factory_(registry->getSuite(current, &suite_name_));
61  if(factory_ != nullptr) {
62  /* -- schedule run of next suite */
63  context_.command_stack->pushCommand(
64  std::make_shared<CmdNextSuite>(registry, current + 1));
65 
66  /* -- create and schedule the suite if it's not filtered */
67  if(!context_.runner_filter->filterSuite(suite_name_)) {
68  /* -- prepare stack-frame of the suite - suite's result and suite's
69  * path. */
70  context_.object_path->pushName(suite_name_);
71  context_.semantic_stack->push(true);
72 
73  /* -- report entering of the suite */
74  context_.reporter->enterSuite(context_, suite_name_);
75 
76  /* -- schedule finishing of the suite */
77  context_.command_stack->pushCommand(std::make_shared<CmdLeaveSuite>());
78 
79  /* -- The constructor method of the suite may throw and exception.
80  * So I do the creation in a protected environment. */
81  runUserCode(context_, [&](const Context& context_) {
82  SuitePtr suite_(factory_->createSuite(context_));
83  suite_->scheduleRun(context_, suite_);
84  });
85  }
86  }
87 }
