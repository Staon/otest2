53 void CmdRepeatObject::run(
54  const Context& context_) {
55  if(repeater->isNextRun(context_)) {
56  /* -- decorate object's name for next run */
57  std::string decorated_name_(repeater->transformName(context_, name));
58 
59  /* -- schedule myself for next run */
60  context_.command_stack->pushCommand(
61  std::make_shared<CmdRepeatObject>(scenario, repeater, name, parent));
62 
63  /* -- prepare stack-frame of the object - object's result and path. */
64  context_.object_path->pushName(decorated_name_);
65  context_.semantic_stack->push(true);
66 
67  /* -- report entering of the suite */
68  scenario->reportEntering(context_, decorated_name_);
69 
70  /* -- schedule finishing of the suite */
71  context_.command_stack->pushCommand(
72  std::make_shared<CmdLeaveObject>(scenario, decorated_name_));
73 
74  /* -- The constructor method of the suite may throw and exception.
75  * So I do the creation in a protected environment. */
76  runUserCode(context_, [&](const Context& context_) {
77  ObjectScenarioPtr object_(
78  repeater->createObject(context_, decorated_name_, parent));
79  context_.command_stack->pushCommand(
80  std::make_shared<CmdStartUpObject>(object_, scenario, parent, 0));
81  });
82  }
83 }
