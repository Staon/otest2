47 void CmdStartSuite::run(
48  const Context& context_) {
49  /* -- report start of the suite */
50  context_.object_path->pushName(suite->getName());
51  context_.reporter->enterSuite(context_, suite->getName());
52 
53  /* -- prepare suite's result */
54  context_.semantic_stack->push(true);
55 
56  /* -- begin initialization of the suite */
57  suite->startUpSuite(context_);
58  if(!context_.semantic_stack->top()) {
59  /* -- initialization failed */
60  context_.reporter->leaveSuite(context_, suite->getName(), false);
61  context_.object_path->popName();
62  context_.semantic_stack->popAnd();
63  }
64  else {
65  /* -- initialization passed - push the clean-up command */
66  context_.command_stack->pushCommand(
67  std::make_shared<CmdDestroySuite>(suite));
68 
69  /* -- run the suite */
70  context_.command_stack->pushCommand(
71  std::make_shared<CmdNextCase>(suite, 0));
72  }
