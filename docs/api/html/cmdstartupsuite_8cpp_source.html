48 void CmdStartUpSuite::run(
49  const Context& context_) {
50  if(suite->startUpSuite(context_, index)) {
51  /* -- check result of the start-up function */
52  if(context_.semantic_stack->top()) {
53  /* -- the start-up function passed -> schedule next start-up function */
54  context_.command_stack->pushCommand(
55  std::make_shared<CmdStartUpSuite>(suite, index + 1));
56  }
57  else {
58  /* -- the start-up function failed -> schedule clean-up of previously
59  * invoked functions. */
60  if(index > 0)
61  context_.command_stack->pushCommand(
62  std::make_shared<CmdTearDownSuite>(suite, index - 1));
63  }
64  }
65  else {
66  /* -- all start-up functions are finished, schedule body of the suite
67  * and clean-up of the suite */
68  if(index > 0)
69  context_.command_stack->pushCommand(
70  std::make_shared<CmdTearDownSuite>(suite, index - 1));
71  context_.command_stack->pushCommand(std::make_shared<CmdNextCase>(suite, 0));
72  }
73 }
