147 Registry& Registry::instance(
148  const std::string& domain_) {
149  typedef std::map<std::string, Registry> TestDomains;
150  static TestDomains domains_;
151  auto iter_(domains_.find(domain_));
152  if(iter_ == domains_.end()) {
153  auto insert_ret_(domains_.emplace(
154  std::piecewise_construct,
155  std::forward_as_tuple(domain_),
156  std::forward_as_tuple()));
157  assert(insert_ret_.second);
158  iter_ = insert_ret_.first;
159  }
160  return (*iter_).second;
