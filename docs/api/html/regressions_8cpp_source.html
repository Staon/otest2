37 bool RegressionAssertion::compareObjectMark(
38  const std::string& key_,
39  TestMarkPtr test_mark_,
40  bool print_) {
41  assert(test_mark_ != nullptr);
42 
43  const Context& context_(otest2Context());
44 
45  /* -- get test mark from the storage */
46  const std::string full_key_(context_.object_path->getRegressionKey(key_));
47  TestMarkPtr stored_(context_.test_mark_storage->getTestMark(full_key_));
48 
49  /* -- compare the marks */
50  bool equal_(false);
51  if(stored_ != nullptr)
52  equal_ = test_mark_->isEqual(*stored_);
53 
54  /* -- report the assertion */
55  std::ostringstream oss_;
56  if(equal_)
57  oss_ << "regression check '" << full_key_ << "' has passed";
58  else
59  oss_ << "regression check '" << full_key_ << "' has failed";
60  enterAssertion(equal_, oss_.str(), false);
61 
62  /* -- report the difference if the check fails */
63  if(!equal_) {
64  /* -- print current test mark */
65  oss_.str("");
66  test_mark_->printMark(oss_, " ");
67  assertionMessage(equal_, "--------- Current ---------");
68  assertionMessage(equal_, oss_.str());
69 
70  assertionMessage(equal_, "--------- Original ---------");
71  if(stored_ != nullptr) {
72  /* -- print the original mark */
73  oss_.str("");
74  stored_->printMark(oss_, " ");
75  assertionMessage(equal_, oss_.str());
76  }
77 
78  assertionMessage(equal_, "--------- Difference ---------");
79  if(stored_ != nullptr) {
80  /* -- compute the difference */
81  std::vector<TestMark::LinearizedRecord> array1_;
82  std::vector<TestMark::LinearizedRecord> array2_;
83  DiffLogBlocks diff_log_;
84  DiffLogBuilderBlock log_builder_(&diff_log_);
85  test_mark_->computeDiff(*stored_, array1_, array2_, log_builder_);
86 
87  /* -- print the difference */
88  oss_.str("");
89  printTestMarkDiff(oss_, array1_, array2_, diff_log_, 3);
90  assertionMessage(equal_, oss_.str());
91  }
92  else {
93  /* -- the new mark is one big addition */
94  oss_.str("");
95  test_mark_->printMark(oss_, "+ ");
96  assertionMessage(equal_, oss_.str());
97  }
98  }
99  else if(print_) {
100  /* -- print current test mark */
101  std::cout << "test mark key: " << full_key_ << std::endl;
102  test_mark_->printMark(std::cout, "");
103  }
104 
105  return leaveAssertion(equal_);
106 }

108 bool RegressionAssertion::storeObjectMark(
109  const std::string& key_,
110  TestMarkPtr test_mark_) {
111  assert(test_mark_ != nullptr);
112 
113  const Context& context_(otest2Context());
114 
115  /* -- store the mark */
116  const std::string full_key_(context_.object_path->getRegressionKey(key_));
117  context_.test_mark_storage->setTestMark(full_key_, test_mark_);
118 
119  /* -- report the assertion */
120  std::ostringstream oss_;

122  enterAssertion(false, oss_.str(), false);
123  return leaveAssertion(false);
