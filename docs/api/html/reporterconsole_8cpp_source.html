40 void printHR(
41  std::ostream& os_,
42  char rule_char_,
43  const std::string& label_) {
44  /* -- label width */
45  int label_width_(0);
46  if(!label_.empty()) {
47  label_width_ = static_cast<int>(label_.size()) + 2;
48  }
49 
50  /* -- compute length of the left and righ part of the rule */
51  int left_width_((WIDTH - 2 - label_width_) / 2);
52  int right_width_(WIDTH - 2 - label_width_ - left_width_);
53 
54  /* -- print left rule */
55  os_ << ' ';
56  for(int i_(0); i_ < left_width_; ++i_)
57  os_ << rule_char_;
58 
59  /* -- print label */
60  if(!label_.empty()) {
61  os_ << ' ' << label_ << ' ';
62  }
63 
64  /* -- print right rule */
65  for(int i_(0); i_ < right_width_; ++i_)
66  os_ << rule_char_;
67 
68  os_ << std::endl;
69 }

71 void printResultLine(
72  std::ostream& os_,
73  TerminalDriver& driver_,
74  const std::string& label_,
75  bool result_) {
76  int label_width_(static_cast<int>(label_.size()));
77  int free_space_width_(WIDTH - 4 - label_width_ - 8);
78 
79  os_ << " " << label_;
80  for(int i_(0); i_ < free_space_width_; ++i_)
81  os_ << ' ';
82  os_ << '[';
83 
84  if(result_) {
85  driver_.setFGColor(TerminalDriver::GREEN);
86  os_ << "Passed";
87  driver_.cleanAttributes();
88  }
89  else {
90  driver_.setFGColor(TerminalDriver::RED);
91  os_ << "Failed";
92  driver_.cleanAttributes();
93  }
94  os_ << ']' << std::endl;
95 }

97 void printTotalResultLine(
98  std::ostream& os_,
99  const std::string& label_,
100  const std::string& ok_,
101  const std::string& failed_,
102  const std::string& total_) {
103  int label_width_(static_cast<int>(label_.size()));
104  int ok_width_(static_cast<int>(ok_.size()));
105  int failed_width_(static_cast<int>(failed_.size()));
106 
107  enum {
108  OK_COLUMN = 20,
109  FAILED_COLUMN = 40,
110  TOTAL_COLUMN = 60,
111  };
112 
113  os_ << " " << label_;
114  int i_(2 + label_width_);
115  for(; i_ < OK_COLUMN; ++i_)
116  os_ << ' ';
117  os_ << ok_;
118  for(i_ += ok_width_; i_ < FAILED_COLUMN; ++i_)
119  os_ << ' ';
120  os_ << failed_;
121  for(i_ += failed_width_; i_ < TOTAL_COLUMN; ++i_)
122  os_ << ' ';
123  os_ << total_ << std::endl;
124 }

229  bool condition_,
230  const std::string& message_,
231  const std::string& file_,
232  int lineno_) {
233  /* -- adjust the statistics */

235 
236  /* -- store the condition for printing of additional messages */

238 
239  /* -- print the assertion status */
240  if(pimpl->verbose || !condition_) {
241  *pimpl->os
