51 void printHR(
52  std::ostream& os_,
53  char rule_char_,
54  const std::string& label_,
55  int indent_) {
56  /* -- label width */
57  int label_width_(0);
58  if(!label_.empty()) {
59  label_width_ = static_cast<int>(label_.size()) + 2;
60  }
61 
62  /* -- compute length of the left and righ part of the rule */
63  const int width_(WIDTH - 2 - INDENT_SPACE * indent_);
64  int left_width_((width_ - label_width_) / 2);
65  int right_width_(width_ - label_width_ - left_width_);
66 
67  /* -- print indentation */
68  printIndent(os_, indent_);
69 
70  /* -- print left rule */
71  os_ << ' ';
72  for(int i_(0); i_ < left_width_; ++i_)
73  os_ << rule_char_;
74 
75  /* -- print label */
76  if(!label_.empty()) {
77  os_ << ' ' << label_ << ' ';
78  }
79 
80  /* -- print right rule */
81  for(int i_(0); i_ < right_width_; ++i_)
82  os_ << rule_char_;
83 
84  os_ << std::endl;
85 }

87 void printResultLine(
88  std::ostream& os_,
89  TerminalDriver& driver_,
90  const std::string& label_,
91  bool result_,
92  int indent_) {
93  int label_width_(static_cast<int>(label_.size()));
94  int free_space_width_(WIDTH - 4 - INDENT_SPACE * indent_ - label_width_ - 8);
95 
96  printIndent(os_, indent_);
97 
98  os_ << " " << label_;
99  for(int i_(0); i_ < free_space_width_; ++i_)
100  os_ << ' ';
101  os_ << '[';
102 
103  if(result_) {
104  driver_.setForeground(*os_.rdbuf(), Color::GREEN);
105  os_ << "Passed";
106  }
107  else {
108  driver_.setForeground(*os_.rdbuf(), Color::RED);
109  os_ << "Failed";
110  }
111  driver_.cleanAttributes(*os_.rdbuf());
112  os_ << ']' << std::endl;
113 }

115 void printTotalResultLine(
116  std::ostream& os_,
117  const std::string& label_,
118  const std::string& ok_,
119  const std::string& failed_,
120  const std::string& total_) {
121  int label_width_(static_cast<int>(label_.size()));
122  int ok_width_(static_cast<int>(ok_.size()));
123  int failed_width_(static_cast<int>(failed_.size()));
124 
125  enum {
126  OK_COLUMN = 20,
127  FAILED_COLUMN = 40,
128  TOTAL_COLUMN = 60,
129  };
130 
131  os_ << " " << label_;
132  int i_(2 + label_width_);
133  for(; i_ < OK_COLUMN; ++i_)
134  os_ << ' ';
135  os_ << ok_;
136  for(i_ += ok_width_; i_ < FAILED_COLUMN; ++i_)
137  os_ << ' ';
138  os_ << failed_;
139  for(i_ += failed_width_; i_ < TOTAL_COLUMN; ++i_)
140  os_ << ' ';
141  os_ << total_ << std::endl;
142 }
