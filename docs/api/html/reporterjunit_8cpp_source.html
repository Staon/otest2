38 std::string formatTimestamp(
39  const TimeSource::time_point& time_) {
40  std::time_t epoche_(std::chrono::system_clock::to_time_t(time_));
41  struct tm broken_;
42  localtime_r(&epoche_, &broken_);
43  std::ostringstream oss_;
44  oss_ << std::setfill('0') << std::dec << std::internal
45  << std::setw(4) << (broken_.tm_year + 1900)
46  << '-'
47  << std::setw(2) << (broken_.tm_mon + 1)
48  << '-'
49  << std::setw(2) << broken_.tm_mday
50  << 'T'
51  << std::setw(2) << broken_.tm_hour
52  << ':'
53  << std::setw(2) << broken_.tm_min
54  << ':'
55  << std::setw(2) << broken_.tm_sec;
56  return oss_.str();
57 }

59 std::string formatDuration(
60  const TimeSource::time_point& start_,
61  const TimeSource::time_point& end_) {
62  std::chrono::duration<double> duration_(end_ - start_);
63  std::ostringstream oss_;
64  oss_ << std::setprecision(3) << std::fixed << duration_.count();
65  return oss_.str();
66 }

183 void ReporterJUnit::enterAssert(
184  const Context& context_,
185  bool condition_,
186  const std::string& message_,
187  const std::string& file_,
188  int lineno_) {
189  if(!condition_) {
190  /* -- update statistics */
191  if(!pimpl->case_first_failure)
192  ++pimpl->case_failures;
193  else
194  pimpl->case_first_failure = true;
195 
196  /* -- make the failure record */
197  auto failure_(pimpl->testcase.append_child("failure"));
198  auto message_attr_(failure_.append_attribute("message"));
199  message_attr_ = message_.c_str();
200  auto line_attr_(failure_.append_attribute("line"));
201  line_attr_ = lineno_;
202  auto file_attr_(failure_.append_attribute("file"));
203  file_attr_ = file_.c_str();
204  }
205 }
