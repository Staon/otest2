41 std::string formatTimestamp(
42  const TimeSource::time_point& time_) {
43  std::time_t epoche_(std::chrono::system_clock::to_time_t(time_));
44  struct tm broken_;
45  localtime_r(&epoche_, &broken_);
46  std::ostringstream oss_;
47  oss_ << std::setfill('0') << std::dec << std::internal
48  << std::setw(4) << (broken_.tm_year + 1900)
49  << '-'
50  << std::setw(2) << (broken_.tm_mon + 1)
51  << '-'
52  << std::setw(2) << broken_.tm_mday
53  << 'T'
54  << std::setw(2) << broken_.tm_hour
55  << ':'
56  << std::setw(2) << broken_.tm_min
57  << ':'
58  << std::setw(2) << broken_.tm_sec;
59  return oss_.str();
60 }

62 std::string formatDuration(
63  const TimeSource::time_point& start_,
64  const TimeSource::time_point& end_) {
65  std::chrono::duration<double> duration_(end_ - start_);
66  std::ostringstream oss_;
67  oss_ << std::setprecision(3) << std::fixed << duration_.count();
68  return oss_.str();
69 }

233 void ReporterJUnit::Impl::assertionOpeningMessage(
234  const Context& context_,
235  const AssertBufferAssertData& data_,
236  const std::string& message_) {
237  /* -- make the failure record */
238  if(!data_.condition) {
239  auto& top_(node_stack.back());
240  auto failure_(top_.node.append_child("failure"));
241  if(!hide_location) {
242  auto line_attr_(failure_.append_attribute("line"));
243  line_attr_ = data_.line;
244  auto file_attr_(failure_.append_attribute("file"));
245  file_attr_ = data_.file.c_str();
246  }
247  message << message_;
248  }

354 AssertBufferPtr ReporterJUnit::enterAssert(
355  const Context& context_,
356  bool condition_,
357  const std::string& file_,
358  int lineno_) {
359  if(!condition_) {
360  /* -- update statistics */
361  auto& top_(pimpl->node_stack.back());
362  if(top_.first_failure) {
363  top_.first_failure = false;
364  ++top_.case_failures;
365  }
366  }
367 
368  pimpl->assert_buffer->openAssertion({condition_, file_, lineno_});
369  return pimpl->assert_buffer;
370 }
