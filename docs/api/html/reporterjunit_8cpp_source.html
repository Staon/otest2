39 std::string formatTimestamp(
40  const TimeSource::time_point& time_) {
41  std::time_t epoche_(std::chrono::system_clock::to_time_t(time_));
42  struct tm broken_;
43  localtime_r(&epoche_, &broken_);
44  std::ostringstream oss_;
45  oss_ << std::setfill('0') << std::dec << std::internal
46  << std::setw(4) << (broken_.tm_year + 1900)
47  << '-'
48  << std::setw(2) << (broken_.tm_mon + 1)
49  << '-'
50  << std::setw(2) << broken_.tm_mday
51  << 'T'
52  << std::setw(2) << broken_.tm_hour
53  << ':'
54  << std::setw(2) << broken_.tm_min
55  << ':'
56  << std::setw(2) << broken_.tm_sec;
57  return oss_.str();
58 }

60 std::string formatDuration(
61  const TimeSource::time_point& start_,
62  const TimeSource::time_point& end_) {
63  std::chrono::duration<double> duration_(end_ - start_);
64  std::ostringstream oss_;
65  oss_ << std::setprecision(3) << std::fixed << duration_.count();
66  return oss_.str();
67 }

256 void ReporterJUnit::enterAssert(
257  const Context& context_,
258  bool condition_,
259  const std::string& message_,
260  const std::string& file_,
261  int lineno_) {
262  if(!condition_) {
263  /* -- update statistics */
264  auto& top_(pimpl->node_stack.back());
265  if(top_.first_failure) {
266  top_.first_failure = false;
267  ++top_.case_failures;
268  }
269 
270  /* -- make the failure record */
271  auto failure_(top_.node.append_child("failure"));
272  auto message_attr_(failure_.append_attribute("message"));
273  message_attr_ = message_.c_str();
274  if(!pimpl->hide_location) {
275  auto line_attr_(failure_.append_attribute("line"));
276  line_attr_ = lineno_;
277  auto file_attr_(failure_.append_attribute("file"));
278  file_attr_ = file_.c_str();
279  }
280  }
281 }
