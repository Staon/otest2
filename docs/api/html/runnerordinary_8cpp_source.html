133 RunnerResult RunnerOrdinary::runNext() {
134  bool first_command_(true);
135  while(!pimpl->command_stack.empty()) {
136  CommandPtr cmd_(pimpl->command_stack.topCommand());
137 
138  /* -- check whether we should get back into the main loop */
139  int delay_(0);
140  if(!first_command_ && cmd_->shouldWait(pimpl->context, delay_)) {
141  assert(delay_ >= 0);
142  return RunnerResult(true, false, delay_);
143  }
144 
145  /* -- run the command */
146  first_command_ = false;
147  pimpl->command_stack.popCommand();
148  cmd_->run(pimpl->context);
149  }
150 
151  /* -- check whether the stack is finished - there is only one value
152  * meant the result of the test. */
153  assert(pimpl->semantic_stack.isFinished());
154 
155  /* -- no other work - get the test result and stop */
156  return RunnerResult(false, pimpl->semantic_stack.top(), -1);
157 }
