80 ScenarioPtr ScenarioCase::filterScenario(
81  TagsStack& tags_,
82  ScenarioContainerPtr parent_,
83  const RunnerFilter& filter_) const {
84  /* -- add my name nad tags into the whole path */
85  tags_.pushTags(pimpl->name, pimpl->tags);
86 
87  /* -- if the object is not filtered append it into the parent container */
88  if(!filter_.filterPath(tags_)) {
89  /* -- the object is supposed to be run add myself into the parent
90  * container. */
91  parent_->appendScenario(
92  pimpl->name,
93  std::make_shared<ScenarioCase>(
94  pimpl->name, pimpl->tags, pimpl->repeater_factory));
95  }
96 
97  /* -- pop me */
98  tags_.popTags();
99 
100  return parent_;
101 }
