111 ScenarioPtr ScenarioRoot::filterScenario(
112  ObjectPath& path_,
113  TagsStack& tags_,
114  ScenarioContainerPtr parent_,
115  const RunnerFilter& name_filter_,
116  const TagFilter& tag_filter_) const {
117  path_.pushName(pimpl->name);
118 
119  ScenarioContainerPtr root_(std::make_shared<ScenarioRoot>(pimpl->name));
120  for(const auto& item_ : pimpl->children) {
121  item_.second->filterScenario(
122  path_, tags_, root_, name_filter_, tag_filter_);
123  }
124 
125  path_.popName();
126 
127  return root_;
128 }
