89 ScenarioPtr ScenarioSuite::filterScenario(
90  ObjectPath& path_,
91  TagsStack& tags_,
92  ScenarioContainerPtr parent_,
93  const RunnerFilter& name_filter_,
94  const TagFilter& tag_filter_) const {
95  /* -- add myself into the stacks */
96  path_.pushName(pimpl->name);
97  tags_.pushTags(pimpl->tags);
98 
99  /* -- create new suite container and filter children */
100  ScenarioContainerPtr suite_(std::make_shared<ScenarioSuite>(
101  pimpl->name, pimpl->tags, pimpl->repeater_factory));
102  for(auto iter_ : pimpl->order) {
103  (*iter_).second->filterScenario(
104  path_, tags_, suite_, name_filter_, tag_filter_);
105  }
106 
107  /* -- append myself if there are some not-filtered children */
108  if(!suite_->isEmpty()) {
109  parent_->appendScenario(pimpl->name, suite_);
110  }
111 
112  /* -- remove myself from the object path */
113  path_.popName();
114  tags_.popTags();
115 
116  return parent_;
117 }
