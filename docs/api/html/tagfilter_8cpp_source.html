407 bool TagGlob::matches(
408  const std::vector<Tags>& tags_stack_) const {
409  /* -- non-deterministic automaton */
410  std::set<int> states_{0};
411  for(const auto& tags_ : tags_stack_) {
412  /* -- compute new generation of states */
413  std::set<int> new_states_;
414  while(!states_.empty()) {
415  int state_index_(*states_.begin());
416  states_.erase(states_.begin());
417  if(state_index_ < globs.size())
418  globs[state_index_]->transition(
419  states_, new_states_, state_index_, tags_);
420  }
421 
422  /* -- switch new generation of states */
423  states_.swap(new_states_);
424  new_states_.clear();
425  }
426 
427  /* -- if the final state has been reached */
428  return states_.find(static_cast<int>(globs.size())) != states_.end();
429 }

500 Token Lexan::getNextToken() {
501  /* -- parsed token from previous run */
502  if(prepared.type != TokenType::NONE) {
503  Token retval_(prepared);
504  prepared.type = TokenType::NONE;
505  return retval_;
506  }
507 
508  while(true) {
509  int c_(is->get());
510  ++location;
511 
512  switch(state) {
513  case State::START:
514  if(std::istream::traits_type::not_eof(c_)) {
515  if(std::isspace(c_)) {
516  /* -- ignore spaces */
517  }
518  else if(std::isalpha(c_) || c_ == '_') {
519  state = State::TAG;
520  value.push_back(c_);
521  }
522  else if(c_ == '<') {
523  state = State::CLASS;
524  }
525  else if(c_ == '!') {
526  return {TokenType::NOT, "!"};
527  }
528  else if(c_ == '|') {
529  state = State::OR;
530  }
531  else if(c_ == '&') {
532  state = State::AND;
533  }
534  else if(c_ == '(') {
535  return {TokenType::LEFT, "("};
536  }
537  else if(c_ == ')') {
538  return {TokenType::RIGHT, ")"};
539  }
540  else if(c_ == ':') {
541  state = State::QUAD;
542  }
543  else if(c_ == '*') {
544  state = State::STAR;
545  }
546  else if(c_ == '[') {
547  return {TokenType::LBRACK, "["};
548  }
549  else if(c_ == ']') {
550  return {TokenType::RBRACK, "]"};
551  }
552  else {
553  /* -- invalid character */
554  throw TagExpressionException(location, c_);
555  }
556  break;
557  }
558  else {
559  /* -- end of the string -> OK here */
560  return {TokenType::END, ""};
561  }
562 
563  case State::TAG:
564  if(std::istream::traits_type::not_eof(c_)) {
565  if(std::isspace(c_)) {
566  state = State::START;
567  return resetValue(TokenType::TAG);
568  }
569  else if(std::isalnum(c_) || c_ == '_' || c_ == '-' || c_ == '.') {
570  value.push_back(c_);
571  }
572  else if(c_ == '<') {
573  state = State::CLASS;
574  return resetValue(TokenType::TAG);
575  }
576  else if(c_ == '!') {
577  prepared = {TokenType::NOT, "!"};
578  state = State::START;
579  return resetValue(TokenType::TAG);
580  }
581  else if(c_ == '|') {
582  state = State::OR;
583  return resetValue(TokenType::TAG);
584  }
585  else if(c_ == '&') {
586  state = State::AND;
587  return resetValue(TokenType::TAG);
588  }
589  else if(c_ == '(') {
590  prepared = {TokenType::LEFT, "("};
591  state = State::START;
592  return resetValue(TokenType::TAG);
593  }
594  else if(c_ == ')') {
595  prepared = {TokenType::RIGHT, ")"};
596  state = State::START;
597  return resetValue(TokenType::TAG);
598  }
599  else if(c_ == ':') {
600  state = State::QUAD;
601  return resetValue(TokenType::TAG);
602  }
603  else if(c_ == '*') {
604  state = State::STAR;
605  return resetValue(TokenType::TAG);
606  }
607  else if(c_ == '[') {
608  prepared = {TokenType::LBRACK, "["};
609  state = State::START;
610  return resetValue(TokenType::TAG);
611  }
612  else if(c_ == ']') {
613  prepared = {TokenType::RBRACK, "]"};
614  state = State::START;
615  return resetValue(TokenType::TAG);
616  }
617  else {
618  throw TagExpressionException(location, c_);
619  }
620  break;
621  }
622  else {
623  /* -- end of string -> return the parsed tag */
624  state = State::START;
625  return resetValue(TokenType::TAG);
626  }
627 
628  case State::CLASS:
629  if(std::istream::traits_type::not_eof(c_)) {
630  if(std::isalpha(c_)) {
631  value.push_back(c_);
632  }
633  else if(c_ == '>') {
634  state = State::START;
635  return resetValue(TokenType::CLASS);
636  }
637  else {
638  throw TagExpressionException(location, c_);
639  }
640  break;
641  }
642  else {
643  throw TagExpressionException("unexpected end of the expression");
644  }
645 
646  case State::OR:
647  if(std::istream::traits_type::not_eof(c_)) {
648  if(c_ == '|') {
649  state = State::START;
650  return {TokenType::OR, "||"};
651  }
652  else {
653  throw TagExpressionException(location, c_);
654  }
655  }
656  else {
657  throw TagExpressionException("unexpected end of the expression");
658  }
659 
660  case State::AND:
661  if(std::istream::traits_type::not_eof(c_)) {
662  if(c_ == '&') {
663  state = State::START;
664  return {TokenType::AND, "&&"};
665  }
666  else {
667  throw TagExpressionException(location, c_);
668  }
669  }
670  else {
671  throw TagExpressionException("unexpected end of the expression");
672  }
673 
674  case State::QUAD:
675  if(std::istream::traits_type::not_eof(c_)) {
676  if(c_ == ':') {
677  state = State::START;
678  return {TokenType::QUAD, "::"};
679  }
680  else {
681  throw TagExpressionException(location, c_);
682  }
683  }
684  else {
685  throw TagExpressionException("unexpected end of the expression");
686  }
687 
688  case State::STAR:
689  if(std::istream::traits_type::not_eof(c_)) {
690  if(std::isspace(c_)) {
691  /* -- ignore spaces */
692  state = State::START;
693  return {TokenType::STAR, "*"};
694  }
695  else if(std::isalpha(c_) || c_ == '_') {
696  state = State::TAG;
697  value.push_back(c_);
698  return {TokenType::STAR, "*"};
699  }
700  else if(c_ == '<') {
701  state = State::CLASS;
702  return {TokenType::STAR, "*"};
703  }
704  else if(c_ == '!') {
705  prepared = {TokenType::NOT, "!"};
706  state = State::START;
707  return {TokenType::STAR, "*"};
708  }
709  else if(c_ == '|') {
710  state = State::OR;
711  return {TokenType::STAR, "*"};
712  }
713  else if(c_ == '&') {
714  state = State::AND;
715  return {TokenType::STAR, "*"};
716  }
717  else if(c_ == '(') {
718  prepared = {TokenType::LEFT, "("};
719  state = State::START;
720  return {TokenType::STAR, "*"};
721  }
722  else if(c_ == ')') {
723  prepared = {TokenType::RIGHT, ")"};
724  state = State::START;
725  return {TokenType::STAR, "*"};
726  }
727  else if(c_ == ':') {
728  state = State::QUAD;
729  return {TokenType::STAR, "*"};
730  }
731  else if(c_ == '*') {
732  state = State::START;
733  return {TokenType::DOUBLE, "**"};
734  }
735  else if(c_ == '[') {
736  prepared = {TokenType::LBRACK, "["};
737  state = State::START;
738  return resetValue(TokenType::STAR);
739  }
740  else if(c_ == ']') {
741  prepared = {TokenType::RBRACK, "]"};
742  state = State::START;
743  return resetValue(TokenType::STAR);
744  }
745  else {
746  /* -- invalid character */
747  throw TagExpressionException(location, c_);
748  }
749  }
750  else {
751  state = State::START;
752  return {TokenType::STAR, "*"};
753  }
754 
755  default:
756  assert(false);
757  break;
758  }
759  }

823 TagGlobPtr SyntaxParser::glob() {
824  TagGlobPtr glob_(std::make_shared<TagGlob>());
825 
826  /* -- empty glob -> accept all */
827  Token l_(lookAhead());
828  if(l_.type == TokenType::END) {
829  glob_->appendPart(std::make_shared<TagGlobPartDouble>());
830  return glob_;
831  }
832 
833  /* -- sequence of expressions, stars or double stars */
834  while(true) {
835  if(l_.type == TokenType::STAR) {
836  glob_->appendPart(std::make_shared<TagGlobPartStar>());
837  compare(TokenType::STAR);
838  }
839  else if(l_.type == TokenType::DOUBLE) {
840  glob_->appendPart(std::make_shared<TagGlobPartDouble>());
841  compare(TokenType::DOUBLE);
842  }
843  else if(l_.type == TokenType::LBRACK) {
844  compare(TokenType::LBRACK);
845  TagExpressionPtr expr_(expression());
846  compare(TokenType::RBRACK);
847  l_ = lookAhead();
848  if(l_.type == TokenType::STAR) {
849  compare(TokenType::STAR);
850  glob_->appendPart(std::make_shared<TagGlobPartExpression>(
851  expr_, TagGlobPartExpression::ZERO_MORE));
852  }
853  else {
854  glob_->appendPart(std::make_shared<TagGlobPartExpression>(
855  expr_, TagGlobPartExpression::ONCE_MORE));
856  }
857  }
858  else {
859  TagExpressionPtr expr_(expression());
860  glob_->appendPart(std::make_shared<TagGlobPartExpression>(
