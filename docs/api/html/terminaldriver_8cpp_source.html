140 void DriverTerminfo::setFGColor(
141  std::ostream* os_,
142  TerminalDriver::Color color_) {
143  if(!setaf.empty()) {
144  /* -- ANSI colors */
145  int converted_(0);
146  switch(color_) {
147  case TerminalDriver::RED:
148  converted_ = 1;
149  break;
150  case TerminalDriver::GREEN:
151  converted_ = 2;
152  break;
153  default:
154  assert(0);
155  }
156  *os_ << tparm(setaf.c_str(), converted_);
157  }
158  else {
159  /* -- old color system */
160  int converted_(0);
161  switch(color_) {

165  case TerminalDriver::GREEN:
166  converted_ = 2;
167  break;
168  default:
169  assert(0);

171  *os_ << tparm(setf.c_str(), converted_);
172  }

198 TerminalDriver::Impl::Impl(
199  std::ostream* os_) :
200  os(os_) {
201  assert(os != nullptr);
202 
203  /* -- Decide the file handle. Now only standard output and standard
204  * error output are considered being terminals. */
205  int handle_(-1);
206  if(os == &std::cout)
207  handle_ = 1;
208  else if(os == &std::cerr)
209  handle_ = 2;
210 
211  /* -- initialize the terminfo */
212  if(handle_ >= 0 && isatty(handle_)) {
213  int err_;
214  int info_(setupterm(nullptr, handle_, &err_));
215  if(info_ == OK && err_ == 1) {
216  const char* setaf_(tigetstr("setaf"));
217  const char* setf_(tigetstr("setf"));
218  const char* sgr0_(tigetstr("sgr0"));
219  if(sgr0_ > reinterpret_cast<char*>(0)
220  && (setaf_ > reinterpret_cast<char*>(0)
221  || setf_ > reinterpret_cast<char*>(0))) {
222  /* -- the capabilities are correctly read, create the driver */
223  driver.reset(new DriverTerminfo(
224  sgr0_,
225  (setaf_ > reinterpret_cast<char*>(0))?setaf_:"",
226  (setf_ > reinterpret_cast<char*>(0))?setf_:""));
227  return;
228  }
229  }
230  }
231 
232  /* -- In all other cases set the null driver up. */
233  driver.reset(new DriverNull);
234 }
