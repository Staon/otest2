34  int right_line_(0);
35  TestMarkPrinter left_printer_(&left_, left_line_);

37  bool trailing_context_(false);
38 
39  for(const auto& diff_ : diff_log_) {
40  /* -- trailing context of previous change */
41  if(trailing_context_) {
42  for(int beg_(left_line_);
43  left_line_ < beg_ + context_ && left_line_ < diff_.left_begin;
44  )
45  left_printer_.printLine(os_, " ");
46  }
47  else
48  trailing_context_ = true;
49 
50  /* -- skip unchanged lines */
51  if(left_line_ < diff_.left_begin - context_)
52  os_ << " ......... \n";
53  for(; left_line_ < diff_.left_begin - context_;)
54  left_printer_.skipLine(os_);
55  for(; right_line_ < diff_.right_begin;)
56  right_printer_.skipLine(os_);
57 
58  /* -- print the context lines */
59  for(; left_line_ < diff_.left_begin;)
60  left_printer_.printLine(os_, " ");
61 
62  /* -- print the changes */
63  for(; right_line_ < diff_.right_end;)
64  right_printer_.printLine(os_, "- ");
65  for(; left_line_ < diff_.left_end;)
66  left_printer_.printLine(os_, "+ ");
67  }
68 
69  /* -- print trailing context */
70  if(trailing_context_) {
71  bool gap_mark_(true);
72  for(int beg_(left_line_); left_line_ < beg_ + context_;) {
73  if(!left_printer_.printLine(os_, " ")) {
74  gap_mark_ = false;
75  break;
76  }
77  }
