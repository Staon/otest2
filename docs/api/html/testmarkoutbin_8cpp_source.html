84 void TestMarkOutBin::writeFloat(
85  long double value_) {
86  writeTag(TestMarkOutBinTag::FLOAT);
87 
88  /* -- There is no platform independent way how to store floating points
89  * in binary form. Hence, I print the float in the hexadecimal
90  * format and I store the float as a string. */
91  std::ostringstream oss_;
92  oss_ << std::hexfloat << value_;
93  const std::string& str_value_(oss_.str());
94 
95  /* -- write the string */
96  const std::uint64_t length_(str_value_.size());
97  assert(length_ <= std::numeric_limits<std::uint8_t>::max());
98  const std::uint8_t be_length_(static_cast<std::uint8_t>(length_));
99  writeBinaryData(sizeof(be_length_), reinterpret_cast<const std::uint8_t*>(&be_length_));
100  writeBinaryData(length_, reinterpret_cast<const std::uint8_t*>(str_value_.c_str()));
101 }
