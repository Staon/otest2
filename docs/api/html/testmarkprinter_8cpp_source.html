70 void TestMarkPrinter::Impl::printTop(
71  std::ostream& os_,
72  bool open_) {
73  assert(!stack.empty());
74 
75  auto& top_(stack.back());
76  if(!top_.skip) {
77  std::ostringstream oss_;
78  oss_ << top_.prefix;
79  for(int i_(0); i_ < top_.indent; ++i_)
80  oss_ << " ";
81  if(open_) {
82  if(!top_.mark->label.empty())
83  oss_ << top_.mark->label << ": ";
84  top_.mark->me->printOpen(os_, oss_.str());
85  }
86  else
87  top_.mark->me->printClose(os_, oss_.str());
88  }

91 bool TestMarkPrinter::Impl::handleItem(
92  std::ostream& os_,
93  const std::string& prefix_,
94  bool skip_) {
95  assert(array != nullptr);
96 
97  /* -- out of the end */
98  if(index >= array->size()) {
99  /* -- finish all opened nodes */
100  if(!stack.empty()) {
101  printTop(os_, false);
102  stack.pop_back();
103  return true;
104  }
105 
106  /* -- empty stack, empty input sequence => final end */
107  return false;
108  }
109 
110  auto& curr_(array->at(index));
111  if(index == 0) {
112  /* -- first item */
113  stack.push_back({&curr_, prefix_, 0, skip_});
114  printTop(os_, true);
115  }
116  else {
117  assert(!stack.empty());
118 
119  /* -- level(s) up from the tree */
120  auto* top_(&stack.back());
121  if(top_->mark->level + 1 != curr_.level) {
122  printTop(os_, false);
123  stack.pop_back();
124  assert(!stack.empty());
125  return true;
126  }
127 
128  /* -- level down into the tree */
129  stack.push_back({&curr_, prefix_, top_->indent + 1, skip_});
130  printTop(os_, true);
131  }
132 
133  /* -- move to next one */
134  ++index;
135 
136  return true;
137 }
