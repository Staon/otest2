55 std::string::size_type searchForSeparator(
56  const std::string& line_) {
57  bool escaped_(false);
58  const std::string::size_type length_(line_.size());
59  for(std::string::size_type i_(0); i_ < length_; ++i_) {
60  char c_(line_[i_]);
61  if(escaped_) {
62  escaped_ = false;
63  }
64  else {
65  if(c_ == '\\') {
66  escaped_ = true;
67  }
68  else {
69  if(c_ == ':')
70  return i_;
71  }
72  }
73  }
74  return std::string::npos;
75 }

77 std::string escapeKey(
78  const std::string& unescaped_key_) {
79  std::ostringstream oss_;
80  for(char c_ : unescaped_key_) {
81  if(c_ == ':' || c_ == '\\' || std::isspace(c_))
82  oss_.put('\\');
83  oss_.put(c_);
84  }
85  return oss_.str();
86 }

131 TestMarkStorage::Impl::Impl(
132  TestMarkFactory* factory_,
133  const std::string& storage_file_) :
134  factory(factory_),
135  storage_file(storage_file_),
136  storage(),
137  changed(false) {
138  assert(factory != nullptr);
139 
140  /* -- read the test marks from the file */
141  std::ifstream ifs_(storage_file.c_str());
142  while(!!ifs_) {
143  std::ostringstream oss_;
144  ifs_.get(*oss_.rdbuf()); /* -- get the whole line */
145  ifs_.get(); /* -- skip the newline */
146  std::string line_(oss_.str());
147 
148  /* -- ignore empty lines */
149  if(isLineEmpty(line_))
150  continue;
151 
152  /* -- parse the key */
153  auto sep_index_(searchForSeparator(line_));
154  if(sep_index_ <= 0 || sep_index_ == std::string::npos) /* -- wrong line */
155  throw ExcTestMarkIn("invalid format of the test mark storage file " + storage_file);
156  std::string key_(line_.data(), line_.data() + sep_index_);
157 
158  /* -- parse the test mark */
159  std::istrstream iss_(
160  line_.data() + sep_index_ + 1, line_.size() - sep_index_ - 1);
161  Base64IStream base64i_(&iss_);
162  Bzip2IStream bzip2i_(&base64i_);
163  TestMarkInBinIOS reader_(&bzip2i_);
164  TestMarkPtr testmark_(TestMarkInBinIOS::deserialize(*factory, reader_));
165 
166  /* -- insert into the map */
167  storage.insert({unescapeKey(key_), testmark_});
168  }
169 }
